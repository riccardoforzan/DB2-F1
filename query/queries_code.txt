SELECT (COUNT(DISTINCT ?drive) as ?pole_positions) where { 
    
    ?driver person:first_name "Lewis"^^xsd:string . 
    ?driver person:last_name "Hamilton"^^xsd:string .

    ?drive f1:during ?race_weekend . 
    
    ?race_weekend f1:year "2021"^^xsd:int .
            
}
